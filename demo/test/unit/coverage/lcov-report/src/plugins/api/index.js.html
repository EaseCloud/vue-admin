 * 支持 vue-resource 风格的 api 调用
 * 通过 vm.api('model') 可以获取到 vue-resource 方式获取到的 REST-resource 对象
 * 提供相同的接口

import urljoin from 'url-join'
import axios from 'axios'
import config from '../../config'
 
const defaultRoot = config.api_root || '/api'
const defaultAdjust = config.api_model_adjust || 'underscore'
const defaultFormat = config.api_format || '{/id}{/action}/'
 
// All http methods: https://developer.mozilla.org/en-US/docs/Web/HTTP/Methods
const httpMethodsSafe = ['get', 'head', 'connect', 'options', 'trace']
const httpMethodsUnsafe = ['post', 'put', 'delete', 'patch']
const httpMethods = [...httpMethodsSafe, ...httpMethodsUnsafe]
 
function getModelUrl (model, adjust = defaultAdjust, format = defaultFormat) {
  let modelUrl = model
  if (adjust === 'underscore') {
    modelUrl = model.replace(/([A-Z])/g, $1 => `_${$1.toLowerCase()}`).replace(/^_/, '')
  } else if (adjust === 'off') {
    // pass
  } else {
    throw new Error('不支持的 api_model_adjust 选项，可用选项：off | underscore')
  }
  return `${modelUrl}${format}`
}

/**
 * 从 vue-resource 迁移，当 Resource 方法调用 http 方法的时候，传入参数的归一化
 * 分别将参数切分到 payload 里面以及 params/query 里面
 * @param action 传入 Object<method, params>
 * @param args 传入 resource.method(...arguments) 的所有参数
 * @returns {*}
 */
function opts (action, args) {
  const options = Object.assign({}, action)
  let params = {}
  let body
  switch (args.length) {
    case 2:
      params = args[0]
      body = args[1]
      break
    case 1:
      if (httpMethodsUnsafe.indexOf(options.method.toLowerCase()) > -1) {
        body = args[0]
      } else {
        params = args[0]
      }
      break
    case 0:
      break
    default:
      throw new Error('Expected up to 2 arguments [params, body], got ' + args.length + ' arguments')
  }
  options.body = body
  options.params = Object.assign({}, options.params, params)
  return options
}
 
class RestResource {
  constructor (model, root = defaultRoot, adjust = defaultAdjust, format = defaultFormat) {
    const _this = this
    this.model = model
    this.url = urljoin(root, getModelUrl(model, root, adjust, format))
    httpMethods.forEach(method => {
      this[method] = method => {
        _this.request(method.toUpperCase(), ...arguments)
      }
    })
  }
 
  request (method = 'GET') {
    const args = arguments.slice(1)
    const { body, params } = opts({ method }, args)

  }
}
 
export default {
  install (Vue, options = {}) {
    Vue.mixin({
      computed: {
        axios: () => axios.create(config.axios_options)
      },
      methods: {
        api: model => {
          return new RestResource(model || this.model)
        }
      }
    })
  }
}
